```python
# 定义数值列（和你的数据完全匹配）
numeric_cols = [
    # 污染物数据

# ================= 步骤1：基础异常值替换（0/无 → NaN）==================
# 如果你确认0是有效值（比如降水量0mm是正常的），可以把0从列表里删掉
df = df.replace([0, '无'], np.nan)

# ================= 步骤2：核心：IQR法识别离群异常值（行业通用标准）==================
# IQR原理：异常值 = 小于 Q1-1.5*IQR 或 大于 Q3+1.5*IQR
# IQR=上四分位数(Q3)-下四分位数(Q1)，1.5是通用系数，极端场景可以改成3
def outlier_to_nan(series):
    Q1 = series.quantile(0.25)
    Q3 = series.quantile(0.75)
    IQR = Q3 - Q1
    # 异常值边界
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    # 异常值替换为NaN
    return series.mask((series < lower_bound) | (series > upper_bound), np.nan)

# 对所有数值列执行异常值识别替换
df[numeric_cols] = df[numeric_cols].apply(outlier_to_nan)

# ================= 步骤3：处理前统计（异常值+缺失值）==================
print("="*60)
print("【处理前】每列异常值/缺失值数量（已把异常值转为NaN）：")
```

```python
#===============时序数据集构建==================

# 1. 提取特征与目标数据
data_X = df[select_features].values
data_y = df[target_col].values.reshape(-1, 1)

# 2. 数据标准化（消除量纲影响）
scaler_X = StandardScaler()
scaler_y = StandardScaler()
X_scaled = scaler_X.fit_transform(data_X)
y_scaled = scaler_y.fit_transform(data_y)

# 3. 滑动窗口构建时序样本
def create_sequences(X, y, look_back, pred_steps):
    """
    构建时序样本
    :param X: 输入特征 (n_samples, n_features)
    :param y: 预测目标 (n_samples, 1)
    :param look_back: 历史回溯步长
    :param pred_steps: 预测步长
    :return: X_seq (n, look_back, n_features), y_seq (n, pred_steps)
    """
    X_seq, y_seq = [], []
    max_idx = len(X) - look_back - pred_steps + 1
    for i in range(max_idx):
        X_seq.append(X[i:i+look_back, :])
        y_seq.append(y[i+look_back:i+look_back+pred_steps, 0])
    return np.array(X_seq), np.array(y_seq)

# 生成序列数据
X_seq, y_seq = create_sequences(X_scaled, y_scaled, LOOK_BACK, PRED_STEPS)
print(f"序列样本形状: X={X_seq.shape}, y={y_seq.shape}")

# 4. 时序划分训练集与测试集（按时间顺序，禁止随机划分）
train_size = int(len(X_seq) * 0.8)
X_train, X_test = X_seq[:train_size], X_seq[train_size:]
y_train, y_test = y_seq[:train_size], y_seq[train_size:]
print(f"训练集: X={X_train.shape}, y={y_train.shape}")
print(f"测试集: X={X_test.shape}, y={y_test.shape}")

# 5. 构建PyTorch Dataset与DataLoader
class AirDataset(Dataset):
    def __init__(self, X, y):
        self.X = torch.tensor(X, dtype=torch.float32)
        self.y = torch.tensor(y, dtype=torch.float32)
    
    def __len__(self):
        return len(self.X)
    
    def __getitem__(self, idx):
        return self.X[idx], self.y[idx]

train_dataset = AirDataset(X_train, y_train)
test_dataset = AirDataset(X_test, y_test)
train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE, shuffle=True)
test_loader = DataLoader(test_dataset, batch_size=BATCH_SIZE, shuffle=False)
```

```python
#===============PyTorch LSTM 多步预测模型（优化过拟合）==================
class LSTMPredictor(nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, pred_steps, dropout=0.3):  # 增加Dropout从0.2到0.3
        super(LSTMPredictor, self).__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        # LSTM层
        self.lstm = nn.LSTM(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            dropout=dropout if num_layers>1 else 0
        )
        #  dropout层与全连接输出层
        self.dropout = nn.Dropout(dropout)
        self.fc = nn.Linear(hidden_size, pred_steps)
    
    def forward(self, x):
        lstm_out, (hidden, cell) = self.lstm(x)
        out = self.dropout(lstm_out[:, -1, :])
        out = self.fc(out)
        return out

# 初始化模型（优化：隐藏层从64减到32，降低过拟合）
input_size = len(select_features)
model = LSTMPredictor(
    input_size=input_size,
    hidden_size=32,  # 从64减到32
    num_layers=NUM_LAYERS,
    pred_steps=PRED_STEPS,
    dropout=0.3  # 增加Dropout
).to(device)

# 损失函数与优化器（增加L2正则化 weight_decay）
criterion = nn.MSELoss()
optimizer = torch.optim.Adam(model.parameters(), lr=LEARNING_RATE, weight_decay=1e-4)  # 增加L2正则
# 学习率衰减
scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(optimizer, 'min', patience=5, factor=0.5)
# 早停机制
early_stop_patience = 10
best_test_loss = float('inf')
early_stop_counter = 0
```

```python
# =============== 终极字体修复：Ubuntu 24.04 零警告 ===============
import matplotlib.pyplot as plt
# 使用系统自带字体，彻底关闭中文依赖

#=============== 修复早停BUG + 模型训练 ==================
train_loss_list = []
test_loss_list = []

# 重置早停参数（防止之前的变量污染）
best_test_loss = float('inf')
early_stop_counter = 0

print("Start model training...")
for epoch in range(EPOCHS):
    # 训练阶段
    model.train()
    train_loss = 0
    for batch_X, batch_y in tqdm(train_loader, desc=f"Epoch {epoch+1}/{EPOCHS}"):
        batch_X, batch_y = batch_X.to(device), batch_y.to(device)
        outputs = model(batch_X)
        loss = criterion(outputs, batch_y)
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
        
        train_loss += loss.item() * batch_X.size(0)
    avg_train_loss = train_loss / len(train_dataset)
    train_loss_list.append(avg_train_loss)
    
    # 测试阶段
    model.eval()
    test_loss = 0
    with torch.no_grad():
        for batch_X, batch_y in test_loader:
            batch_X, batch_y = batch_X.to(device), batch_y.to(device)
            outputs = model(batch_X)
            loss = criterion(outputs, batch_y)
            test_loss += loss.item() * batch_X.size(0)
    avg_test_loss = test_loss / len(test_dataset)
    test_loss_list.append(avg_test_loss)
    
    # 学习率调整
    scheduler.step(avg_test_loss)
    print(f"Epoch {epoch+1}/{EPOCHS} | Train MSE: {avg_train_loss:.6f} | Test MSE: {avg_test_loss:.6f}")
    
    # 修复后的早停逻辑（绝对不会1轮停止）
    if avg_test_loss < best_test_loss:
        best_test_loss = avg_test_loss
        best_model = model.state_dict()
        early_stop_counter = 0
    else:
        early_stop_counter += 1
    
    # 只有连续10轮不下降才停止
    if early_stop_counter >= early_stop_patience:
        print(f"Early stop at Epoch {epoch+1}")
        break

# 加载最优模型
```
