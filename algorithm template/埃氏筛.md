### C++代码：

```cpp
#define mx = 1e6 + 1
vector<bool> is_prime(mx,true);
vector<int> primes;
void aishi()
{
	is_prime[0]= false;
	is_prime[1] = false;
	for (int i = 2 ; i < mx ; i++)
	{
		if (is_prime[i])
		{
			primes.push_back(i);
		}
		for (long long j = (long long) i * i ; j < mx ; j += i)
		{
			is_prime[j] = false;
		}
	}
}
```

### Python 代码：

```python
mx = 10 ** 6 + 1
primes = []
is_prime = [True] * mx
for i in range(2,mx):
	if is_prime[i]:
		primes.append(i)
		for j in range(i * i,mx,i):
			is_prime[j] = False
primes.extend((mx,mx));
```
