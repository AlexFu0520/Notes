### C++代码

```cpp
long long quick_pow(long long base,long long power,long long mod)
{
	long long ans = 1;
	while (power > 0)
	{
		if (power & 1)
		{
			ans = (base * ans) % mod;
		}
		power >>= 1;
		base = (base * base) % mod;
	}
	return ans;
}
```

### Python代码

```python
def quick_pow(base,power,mod):
	ans = 1
	while power > 0:
		if power & 1:
			ans = (ans * base) % mod
		power >>= 1
		base = (base * base) % mod
	return ans
```
