描述
--

对于给定的无向无根树，第 $i$ 个节点上有一个权值 $w_i$ 。我们定义一条简单路径是好的，当且仅当：路径上的点的点权最小值小于等于 $a$ ，路径上的点的点权最大值大于等于 $b$ 。  
  
保证给定的 $a < b$ ，你需要计算有多少条简单路径是好的。
### 输入描述：

第一行输入三个整数 $n, a, b\left(1 \leq n \leq 5 \times 10^5, 1 \leq a < b \leq 10^9\right)$ 代表节点数、给定的上下限。  
  
第二行输入 $n$ 个整数 $w_1, w_2, \dots, w_n\left(1 \leq w_i \leq 10^9\right)$ 代表每个节点的权值。  
  
此后 $n - 1$ 行，每行输入两个整数 $u, v\left(1 \leq u, v \leq n, u \neq v\right)$ 代表一条无向边连接树上 $u$ 和 $v$ 两个节点。
### 输出描述：

在一行上输出一个整数，代表好路径的条数。
示例1
---

输入：
```txt
5 2 3
5 4 3 3 1
1 2
1 3
3 4
3 5
```

输出：
```txt
4
```

说明：对于这个样例，如下图所示。路径 $2 \to 1 \to 3 \to 5$ 是好的，因为路径点权最小值 $1 \leqq a$ 且点权最大值 $5 \geqq b$ 。

除此之外，以下路径也是好的：
- $1 \to 3 \to 5$ ；
- $3 \to 5$ ；
- $4 \to 3 \to 5$ 。
## 思路：
我们定义一条好路径要满足两个条件：
- A：路径上点权最小值<= a
- B：路径上点权最大值>=b

直接求同时满足这两个条件的路径数比较困难，我们可以通过求它的反面来解决

根据容斥原理：

满足A且满足B的路径数=总路径数-不满足A的路径数-不满足B的路径数+同时不满足A和B的路径数
- 不满足A：路径上所有点权都>a
- 不满足B：路径上所有点权都< b
- 同时不满足A和B：路径上所有点权都满足a < w < b
如何计算满足某种条件的路径数？

我们可以使用并查集
1. 我们只保留不满足条件的节点，将原本的无向图的边用并查集连起来
2. 此时树会被分成若干个连通块，对于一个包含k个节点的连通块，内部能构成的路径数量就是 k*(k + 1) / 2
3. 
## ACcode:
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long ll;

const int MAXN = 5e5+5;
int parent[MAXN]; //父节点数组
ll sz[MAXN]; //连通分量大小:sz[i]表示以i为根的连通分量的节点数

//查找函数（带路径压缩）
int find(int i) {
    if (parent[i] == i) return i; //若i是根节点，直接返回
    //路径压缩：将i的父节点直接设为根，加速后续查找
    return parent[i] = find(parent[i]); 
}

//合并函数（按大小合并）
void unite(int i,int j) {
    int root_i = find(i);
    int root_j = find(j);
    if (root_i != root_j) { //若不在同一连通分量
        if (sz[root_i] < sz[root_j]) swap(root_i,root_j);//小分量合并到大分量，保持树平衡
        parent[root_j] = root_i; //小分量的根指向大分量的根
        sz[root_i] += sz[root_j]; //更新大分量的大小
    }
}

//存储边的两个端点
struct Edge {
    int u,v;
};

ll count(int n,int a,int b,int mode,const vector<int>& w,const vector<Edge>& edges) {
    //初始化并查集
    for (int i = 1 ; i <= n ; ++i) {
        parent[i] = i; //每个节点初始为独立连通分量的根
        sz[i] = 1; //初始大小为1
    }

    //标记活跃节点
    vector<bool> active(n + 1,false);
    for (int i = 1 ; i <= n ; ++i) { 
        if (mode == 1 && w[i - 1] > a) active[i] = true; //权值大于a
        if (mode == 2 && w[i - 1] < b) active[i] = true; //权值小于b
        if (mode == 3 && w[i - 1] > a && w[i - 1] < b) active[i] = true; //权值在（a，b）之间
    }

    //合并活跃节点的连通分量
    for (const auto& e : edges) {
        //只有两边的端点都是活跃节点时 才合并
        if (active[e.u] && active[e.v]) {
            unite(e.u,e.v);
        }
    }

    //统计所有连通分量的路径和
    ll total = 0;
    for (int i = 1 ; i <= n ; ++i) {
        //只统计活跃节点构成的连通分量的根
        if (active[i] && parent[i] == i) {
            //大小为s的连通分量有 s*(s + 1)/2条路径，包含单个节点
            total += sz[i] * (sz[i] + 1) / 2;
        }
    }
    return total;
}

void solve() {
    int n,a,b;
    cin >> n >> a >> b;
    vector<int> w(n);
    for (int i = 0 ; i < n ; ++i) cin >> w[i];
    
    vector<Edge> edges(n - 1);
    for (int i = 0 ; i < n - 1 ; ++i) {
        cin >> edges[i].u >> edges[i].v;
    }

    //计算总路径
    ll total_all = 1ll * n * (n + 1) / 2;

    //计算三种非法路径的数量
    ll count1 = count(n,a,b,1,w,edges);
    ll count2 = count(n,a,b,2,w,edges);
    ll count3 = count(n,a,b,3,w,edges);

    //容斥原理计算答案
    ll ans = total_all - count1 - count2 + count3;

    cout << ans << '\n';
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int T = 1;
    // cin >> T;
    while (T--) {
        solve();   
    } 
    return 0;
}
```
